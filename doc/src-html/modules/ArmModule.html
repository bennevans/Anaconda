001package modules;
002
003import config.ArmConfig;
004import edu.wpi.first.wpilibj.Encoder;
005import edu.wpi.first.wpilibj.PIDController;
006import edu.wpi.first.wpilibj.PIDOutput;
007import edu.wpi.first.wpilibj.Victor;
008
009/**
010 * ArmModule, the class, defines through various methods what the arm on our robot
011 * will be able to do. It has methods that sets the arm in a specific position or in any
012 * position.
013 * @author Ankith Uppunda
014 */
015public class ArmModule extends Module implements PIDOutput {
016    private PIDController controller;
017    private Encoder e;
018    private Victor roller1, roller2, arm1, arm2;
019    public ArmModule(int v1, int v2, int v3, int v4, int edport1, int edport2){
020       controller = new PIDController(ArmConfig.p, ArmConfig.i, ArmConfig.d,e,this);
021       e = new Encoder(edport1, edport2);
022       roller1 = new Victor(v1);
023       roller2 = new Victor(v2);
024       arm1 = new Victor(v3);
025       arm2 = new Victor(v4);
026    }
027/**
028 * turn roller on or off
029 * @param on 
030 */
031    public synchronized void setRoller(boolean on){
032        if(on)
033        {
034            roller1.set(ArmConfig.ROLLER_SPEED);
035            roller2.set(-ArmConfig.ROLLER_SPEED);
036        }
037        
038    }
039    
040 /**
041  * sets the position of the arm to the left
042  */
043    public synchronized void setPosition_left()
044    {
045        setPosition(0);
046    }
047/**
048 * sets the position of the arm to the right
049 */    
050    public synchronized void setPosition_right()
051    {
052        setPosition(180);
053    }
054/**
055 * sets the position of the arm to an angled position
056 */    
057    public synchronized void setPosition_angled()
058    {
059       setPosition(45);
060    }
061/**
062 * sets position of arm to up
063 */    
064    public synchronized void setPosition_up()
065    {
066        setPosition(90);
067    }
068/**
069 * sets p,i,d constants
070 * @param p proportional constant
071 * @param i integral
072 * @param d derivative
073 * 
074 */    
075    public synchronized void setPID(double p, double i, double d){
076        controller.setPID(p, i, d);
077    }
078/**
079 * sets position of the arm
080 * @param setposition 
081 */    
082    public synchronized void setPosition(double setposition)
083    {
084        //TODO calculate correct encoder tick numbers based on angle
085       double setPositionB = (200.0 * setposition)/360 ;
086       controller.setSetpoint(setPositionB);
087    }
088/**
089 * handles arm controller
090 */
091    public void run(){
092        while(true)
093        {
094            if(enabled)
095            {
096                controller.enable();
097            }
098            else
099            {
100                controller.disable();
101            }
102        }
103    }
104/**
105 * resets arm
106 */
107    public void reset(){
108       controller.reset();
109    }
110/**
111 * makes specific value for victors
112 * @param d 
113 * sets the victor value to double d
114 */
115    public void pidWrite(double d) {
116       arm1.set(d);
117       arm2.set(d);
118    }
119/**
120 * makes a string 
121 * @return roller victor values, encoder values, controller errors
122 */
123    public String toString(){
124        return "Roller Value:" + roller1.get() + "Encoder Value:" + e.get() + "PID error:" + controller.getError();
125        
126//you dont need ^
127    }
128    
129    
130}
